---------------------------------------------------------------------------------
Instruções válidas para o Amazon Linux AMI. Para Ubuntu é utilizado o apt-get.
---------------------------------------------------------------------------------

1) Instalar

sudo yum -y install mysql-server
sudo yum -y install mysql

2) Start

sudo service mysqld start

3) Logando como root

mysql -u root

4) Criando o banco de dados

create database livro;
grant all privileges on livro.* to 'livro'@'localhost' identified by 'livro123';
flush privileges;
exit;

5) Login

mysql -u livro -p

digitar senha "livro123"

show databases;
connect livro;

6) Criar tabela

create table usuario (id integer auto_increment, nome varchar(50), email varchar(50), data_cadastro date, primary key(id));
insert into usuario(nome, email, data_cadastro) values ('Ricardo','rlecheta@gmail.com',now());
select * from usuario;

show tables;
describe usuario;

exit;

7) Definir senha root

mysqladmin -u root password

8) Logar como root com senha

mysql -u root -p

9) Dando permissões para um endereço IP

sudo vim /etc/my.cnf

bind-address = *

mysql -u root -p

grant all privileges on livro.* to 'livro'@'ip' identified by 'livro123';
grant all privileges on livro.* to 'livro'@'*' identified by 'livro123';
exit;

sudo service mysqld restart;

10) Backup

* backup
mysqldump -u livro -p livro > livro.sql

* restore
mysql -u livro -p livro < livro.sql

11) Fazer serviço iniciar automaticamente

sudo chkconfig mysqld on
